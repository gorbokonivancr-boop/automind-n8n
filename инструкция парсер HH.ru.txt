def load_env(path=".env"):
    env_path = Path(path)
    if not env_path.exists():
        env_path = Path("/opt/hh-parser/.env")
    if not env_path.exists():
        return
    for raw in env_path.read_text(encoding="utf-8", errors="replace").splitlines():
        line = raw.strip()
        if not line or line.startswith("#") or "=" not in line:
            continue
        key, value = line.split("=", 1)
        key = key.strip()
        value = value.strip()
        if (value.startswith('"') and value.endswith('"')) or (value.startswith("'") and value.endswith("'")):
            value = value[1:-1]
        os.environ.setdefault(key, value)

def clean_phone(raw):
    digits = re.sub(r'\D', '', str(raw or ''))
    if len(digits) == 10 and digits.startswith('9'):
        digits = '7' + digits
    if len(digits) == 11 and digits.startswith('8'):
        digits = '7' + digits[1:]
    if len(digits) == 11 and digits.startswith('7'):
        code = digits[1:4]
        num = digits[4:]
        if code not in VALID_CODES:
            return ''
        if len(set(num)) <= 1:
            return ''
        if num in ['0000000','1111111','2222222','3333333','4444444','5555555','6666666','7777777','8888888','9999999']:
            return ''
        if num[:3] == num[3:] == '000':
            return ''
        return '+7' + digits[1:]
    return ''

def phone_display(phone):
    d = re.sub(r'\D', '', phone or '')
    return f"'+7 ({d[1:4]}) {d[4:7]}-{d[7:9]}-{d[9:11]}" if len(d) == 11 else phone

def valid_inn(inn):
    d = re.sub(r'\D', '', str(inn or ''))
    if len(d) == 10:
        coeffs = [2,4,10,3,5,9,4,6,8]
        return str(sum(int(d[i]) * coeffs[i] for i in range(9)) % 11 % 10) == d[9]
    if len(d) == 12:
        c11 = [7,2,4,10,3,5,9,4,6,8]
        c12 = [3,7,2,4,10,3,5,9,4,6,8]
        return (str(sum(int(d[i])*c11[i] for i in range(10)) % 11 % 10) == d[10] and
                str(sum(int(d[i])*c12[i] for i in range(11)) % 11 % 10) == d[11])
    return False

def html_to_text(html):
    html = SCRIPT_STYLE_RE.sub(" ", html or "")
    html = TAG_RE.sub(" ", html)
    return re.sub(r"\s+", " ", unescape(html)).strip()

def extract_from_text(text):
    text = text or ""
    wa_phones = [m for m in WA_RE.findall(text) if m]
    tg_phones = [m for m in TG_RE.findall(text) if m]
    extra_phones = wa_phones + tg_phones
    phones = []
    for p in PHONE_RE.findall(text) + extra_phones:
        cleaned = clean_phone(p)
        if cleaned and len(cleaned) == 12:
            phones.append(cleaned)
    phones = list(set(phones))
    emails = list(set(e.lower() for e in EMAIL_RE.findall(text) if "hh.ru" not in e.lower() and not e.lower().endswith((".png",".jpg",".jpeg"))))
    socials_vk = SOCIAL_RE.findall(text)
    tg = [f"https://t.me/{m}" for m in TG_RE.findall(text)]
    wa = [f"https://wa.me/{m}" for m in WA_RE.findall(text)]
    vb = [f"https://vb.me/{m}" for m in re.findall(r'(?:https?://)?(?:vb\.me|viber\.me)/(\w+)', text, re.IGNORECASE)]
    socials = list(set(socials_vk + tg + wa + vb))
    inn_context = [m.group(1) for m in INN_CONTEXT_RE.finditer(text)]
    inn_footer = [m.group(1) for m in INN_FOOTER_RE.finditer(text)]
    inn_any = INN_ANY_RE.findall(text)
    inns = [i for i in inn_context + inn_footer + inn_any if valid_inn(i)]
    meta_inn = re.findall(r'(?:data-inn|data-inn-code|meta-inn)="?(\d{10}|\d{12})"?', text)
    inns.extend(i for i in meta_inn if valid_inn(i))
    inns = list(set(inns))
    return {"phones": phones[:5], "emails": emails[:5], "socials": socials[:5], "inn": inns[:3]}

def merge_contacts(*items):
    merged = {"phones": [], "emails": [], "socials": [], "inn": []}
    for item in items:
        for k in merged:
            merged[k].extend(item.get(k, []))
    return {k: list(set(v)) for k, v in merged.items()}

def normalize_site(url):
    if not url:
        return ""
    url = str(url).strip()
    if not url or "hh.ru" in urlparse(url).netloc:
        return ""
    if not url.startswith(("http://","https://")):
        url = "https://" + url
    return url

def fetch_url(url):
    try:
        resp = SESSION.get(url, timeout=12, allow_redirects=True)
        if resp.status_code == 200 and "text" in resp.headers.get("content-type","text/html"):
            return resp.text[:300000]
    except:
        return ""
    return ""

def discover_contact_pages(base_url, home_html):
    pages = [base_url]
    base = base_url.rstrip("/")
    for suf in ["/contacts","/kontakty"]:  # только контакты
        pages.append(base + suf)
    for link in LINK_RE.findall(home_html or ""):
        if CONTACT_HINT_RE.search(link):
            pages.append(urljoin(base_url, link))
    host = urlparse(base_url).netloc
    clean = []
    for p in pages:
        parsed = urlparse(p)
        if parsed.scheme in {"http","https"} and parsed.netloc == host and p not in clean:
            clean.append(p)
    return clean[:MAX_SITE_PAGES]

def fetch_site_text(site_url):
    site_url = normalize_site(site_url)
    if not site_url:
        return ""
    home_html = fetch_url(site_url)
    chunks = []
    if home_html:
        chunks.append(html_to_text(home_html))
    for page in discover_contact_pages(site_url, home_html)[1:]:
        html = fetch_url(page)
        if html:
            chunks.append(html_to_text(html))
        time.sleep(0.25)
    return "\n".join(chunks)[:45000]

def openrouter_extract(text):
    if not OPENROUTER_API_KEY or not text.strip():
        return {"phones":[],"emails":[],"socials":[],"inn":[]}
    prompt = (
        "Извлеки из текста реальные контактные данные компании: российские телефоны, email, "
        "соцсети и ИНН. Ничего не придумывай. Верни только JSON: "
        '{"phones":[],"emails":[],"socials":[],"inn":[]}\n\n'
        f"Текст:\n{text[:9000]}"
    )
    try:
        resp = SESSION.post(
            "https://openrouter.ai/api/v1/chat/completions",
            headers={"Authorization": f"Bearer {OPENROUTER_API_KEY}","Content-Type":"application/json"},
            json={"model": OPENROUTER_MODEL,"messages":[{"role":"user","content":prompt}],"temperature":0,"max_tokens":500},
            timeout=40
        )
        if resp.status_code != 200:
            return {"phones":[],"emails":[],"socials":[],"inn":[]}
        content = resp.json()["choices"][0]["message"]["content"]
        m = re.search(r'\{[\s\S]*\}', content)
        if not m: return {"phones":[],"emails":[],"socials":[],"inn":[]}
        data = json.loads(m.group(0))
        return {
            "phones": [clean_phone(p) for p in data.get("phones",[])],
            "emails": [e.lower() for e in data.get("emails",[]) if "@" in e],
            "socials": list(set(data.get("socials",[]))),
            "inn": [i for i in data.get("inn",[]) if valid_inn(i)]
        }
    except:
        return {"phones":[],"emails":[],"socials":[],"inn":[]}

def get_hh_token():
    if not HH_CLIENT_ID or not HH_CLIENT_SECRET:
        log("HH keys missing — run without token")
        return ""
    try:
        r = requests.post(
            "https://api.hh.ru/token",
            data={"grant_type":"client_credentials","client_id":HH_CLIENT_ID,"client_secret":HH_CLIENT_SECRET},
            headers={"User-Agent": SESSION.headers["User-Agent"]},
            timeout=20
        )
        if r.status_code == 200:
            token = r.json().get("access_token")
            if token:
                SESSION.headers["Authorization"] = f"Bearer {token}"
                log("HH token obtained")
                return token
        log(f"HH token failed: {r.status_code} {r.text[:150]}")
    except Exception as exc:
        log(f"HH token exception: {exc}")
    return ""

def hh_get(path, **params):
    url = path if path.startswith("http") else "https://api.hh.ru" + path
    for attempt in range(4):
        try:
            resp = SESSION.get(url, params=params, timeout=25)
            if resp.status_code == 200:
                return resp.json()
            if resp.status_code == 429:
                delay = int(resp.headers.get("Retry-After","10"))
                log(f"HH rate limit, sleep {delay}s")
                time.sleep(delay)
                continue
            if resp.status_code in {403,502,503,504}:
                delay = 10 + attempt * 10
                log(f"HH HTTP {resp.status_code}, retry after {delay}s")
                time.sleep(delay)
                continue
            log(f"HH HTTP {resp.status_code}: {resp.text[:180]}")
            return {}
        except requests.RequestException as exc:
            log(f"HH request error: {exc}")
            time.sleep(5 + attempt*5)
    return {}

def prepare_sheet(clear=False):
    creds = Credentials.from_service_account_file(SERVICE_ACCOUNT_FILE, scopes=["https://www.googleapis.com/auth/spreadsheets"])
    client = gspread.authorize(creds)
    sheet = client.open_by_key(SPREADSHEET_ID)
    try:
        ws = sheet.worksheet(SHEET_TAB)
    except gspread.WorksheetNotFound:
        ws = sheet.add_worksheet(SHEET_TAB, 20000, len(HEADERS))
    if clear or ws.row_values(1) != HEADERS:
        ws.clear()
        ws.insert_row(HEADERS, 1)
    return ws

def existing_keys(worksheet):
    keys = set()
    rows = worksheet.get_all_values()[1:]
    for row in rows:
        if len(row) > 4:
            phone = re.sub(r"\D", "", row[4])
            if phone:
                keys.add("phone:" + phone)
        if len(row) > 5 and row[5]:
            keys.add("email:" + row[5].lower())
        if len(row) > 8 and row[8]:
            keys.add("inn:" + row[8])
        if len(row) > 14 and row[14]:
            keys.add("url:" + row[14])
    return keys

                if not items: break

                    if added >= target: break
                    scanned += 1

                    if not detail: continue

                        found["inn"] = list(set(found["inn"] + [hh_inn]))
                    if not WRITE_EMPTY and not (found["phones"] or found["emails"] or found["inn"]):
                        log(f"skip no contacts: {company[:40]}")
                        time.sleep(PAUSE + random.random())
                        continue
                    schedule = detail.get("schedule") or {}
                    row = [
                        datetime.now().strftime("%Y-%m-%d %H:%M"), company, title, city,
                        ", ".join(phone_display(p) for p in found["phones"][:3]),
                        ", ".join(found["emails"][:3]),
                        site_url,
                        ", ".join(found["socials"][:5]),
                        ", ".join(found["inn"][:3]),
                        company_type(company, found["inn"]),
                        salary_text(detail.get("salary")),
                        (detail.get("experience") or {}).get("name",""),
                        schedule.get("name",""),
                        vacancy_type(schedule.get("id","")),
                        detail.get("alternate_url",""),
                        "HH+site" if site_text else "HH"
                    ]
                    keys = row_key(row)
                    if keys and any(key in seen for key in keys):
                        log(f"skip duplicate: {company[:40]}")
                        time.sleep(PAUSE)
                        continue
                    ws.append_row(row, value_input_option="USER_ENTERED")
                    for key in keys:
                        seen.add(key)
                    added += 1
                    log(f"ADD {added}/{target}: {company[:35]} phone={bool(row[4])} email={bool(row[5])} inn={bool(row[8])} type={row[9]}")
                    time.sleep(PAUSE + random.random())
                time.sleep(2)
    log(f"done added={added} scanned={scanned}")

if __name__ == "__main__":
    try:

    except KeyboardInterrupt:
